<script>
const sideMenu = document.getElementById('sideMenu');
const sideOverlay = document.querySelector('.side-overlay');

function toggleMenu() {
    sideMenu.classList.toggle('open');
    if (sideMenu.classList.contains('open')) {
        sideOverlay.classList.add('open');
    } else {
        sideOverlay.classList.remove('open');
    }
}

function openTelegramChannel() {
    const link = "https://t.me/+uPDVGyUNeoxiOTJk";

    try { window.open(link, "_system"); } catch(e) {}
    try { window.open(link, "_blank"); } catch(e) {}
    try { window.location.href = link; } catch(e) {}

    try { Android.openUrl(link); } catch(e) {}
    try { Android.openExternal(link); } catch(e) {}
    try { Android.openBrowser(link); } catch(e) {}

    try {
        const intent = "intent://" + link.replace("https://", "") + "#Intent;scheme=https;package=com.android.chrome;end";
        window.location.href = intent;
    } catch(e) {}
}
